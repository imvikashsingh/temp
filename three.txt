import pandas as pd
from google.cloud import storage
from google.cloud import bigquery
import xgboost as xgb
import numpy as np

# Parameters
bucket_name = "your-bucket-name"  # Update with your bucket name
model_file_path = "path/to/model.bst"  # Update with the path within the bucket
local_model_path = "/tmp/model.bst"  # Temporary local path for model
bigquery_table = "your-project-id.your_dataset.your_table"  # Update with your BigQuery table
output_path = "predictions.csv"  # Local output path
feature_columns = ["feature1", "feature2"]  # Replace with actual feature column names

# Function to download model from GCS
def download_model_from_gcs(bucket_name, source_blob_name, destination_file_name):
    storage_client = storage.Client()
    bucket = storage_client.bucket(bucket_name)
    blob = bucket.blob(source_blob_name)
    blob.download_to_filename(destination_file_name)
    return destination_file_name

# Download model from GCS
download_model_from_gcs(bucket_name, model_file_path, local_model_path)

# Load XGBoost model using Booster
xgb_model = xgb.Booster()
xgb_model.load_model(local_model_path)

# Load data from BigQuery
bq_client = bigquery.Client()
query = f"SELECT * FROM `{bigquery_table}`"
pandas_df = bq_client.query(query).to_dataframe()

# Validate feature columns
missing_cols = [col for col in feature_columns if col not in pandas_df.columns]
if missing_cols:
    raise ValueError(f"Missing feature columns in DataFrame: {missing_cols}")

# Prepare features
features = pandas_df[feature_columns].values
print(f"Feature shape: {features.shape}")  # Should be (n_rows, n_features)

# Check model feature count (if supported)
try:
    expected_features = xgb_model.num_features()
    if expected_features and expected_features != len(feature_columns):
        raise ValueError(f"Model expects {expected_features} features, but provided {len(feature_columns)}")
except AttributeError:
    print("Warning: Could not verify model feature count. Ensure feature_columns match training data.")

# Predict row-by-row to avoid shape error
predicted_labels = []
for i, row in enumerate(features):
    # Ensure row is 1D
    row = np.array(row).flatten()
    print(f"Row {i} shape: {row.shape}")  # Should be (n_features,)
    if row.shape[0] != len(feature_columns):
        raise ValueError(f"Row {i} has {row.shape[0]} features, expected {len(feature_columns)}")
    dmatrix = xgb.DMatrix(row.reshape(1, -1))  # Shape (1, n_features)
    prediction = xgb_model.predict(dmatrix)
    # Ensure scalar for binary classification
    if prediction.ndim > 1 and prediction.shape[1] > 1:
        # Multi-class: select class with highest probability
        predicted_labels.append(int(np.argmax(prediction)))
    else:
        # Binary: threshold probability
        predicted_labels.append(int(prediction[0] > 0.5))  # Adjust threshold

# Convert to numpy array and verify shape
predicted_labels = np.array(predicted_labels)
print(f"Predicted labels shape: {predicted_labels.shape}")  # Should be (n_rows,)

# Add predictions to DataFrame
pandas_df["predicted_label"] = predicted_labels

# Select relevant columns
result_df = pandas_df[["id", "predicted_label"]]  # Replace 'id' with your identifier column

# Save results to CSV
result_df.to_csv(output_path, index=False)

# Display first few rows
print(result_df.head())
