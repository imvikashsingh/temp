# Use your existing Python 3.11 image based on Red Hat
FROM your-redhat-python11-image:latest

# Set environment variables (optional, useful for Java apps)
ENV JAVA_HOME=/usr/lib/jvm/java-17-openjdk \
    PATH=$JAVA_HOME/bin:$PATH

# Install Java (OpenJDK 17 in this case, change version if needed)
USER root

RUN yum -y update && \
    yum -y install java-17-openjdk java-17-openjdk-devel && \
    yum clean all

# Verify installations (optional)
RUN java -version && python3 --version

# Set default user if needed
# USER someuser

# Define default command (optional)
CMD ["bash"]
