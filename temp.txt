from google.cloud import bigquery
import pandas as pd

# Initialize BigQuery client
client = bigquery.Client()

# Configuration
project_id = "your-project-id"  # Replace with your GCP project ID
dataset_id = "your_dataset"    # Replace with your BigQuery dataset
model_name = "imported_model"  # Name for the imported model in BigQuery
gcs_model_path = "gs://your-bucket/path/to/model"  # GCS path to exported model
table_id = f"{project_id}.{dataset_id}.your_table"  # BigQuery table with input data

def import_model_from_gcs():
    """Import the exported BQML model from GCS into BigQuery."""
    import_query = f"""
    CREATE OR REPLACE MODEL `{project_id}.{dataset_id}.{model_name}`
    OPTIONS (model_type='BOOSTED_TREE_CLASSIFIER')  -- Adjust model_type as needed
    AS
    SELECT * FROM ML.IMPORT_MODEL(
        MODEL `{project_id}.{dataset_id}.{model_name}`,
        OPTIONS(input_uri='{gcs_model_path}')
    )
    """
    job = client.query(import_query)
    job.result()  # Wait for the query to complete
    print(f"Model {model_name} imported successfully from {gcs_model_path}")

def predict_with_model():
    """Use the imported model to make predictions on data from a BigQuery table."""
    predict_query = f"""
    SELECT
        *
    FROM
        ML.PREDICT(
            MODEL `{project_id}.{dataset_id}.{model_name}`,
            (SELECT * FROM `{table_id}`)
        )
    """
    query_job = client.query(predict_query)
    results = query_job.to_dataframe()  # Convert results to a Pandas DataFrame
    return results

def main():
    # Step 1: Import the model from GCS
    import_model_from_gcs()
    
    # Step 2: Make predictions using the imported model
    predictions = predict_with_model()
    
    # Step 3: Display or process the predictions
    print("Predictions:")
    print(predictions.head())
    
    # Optionally, save predictions to a new BigQuery table
    output_table = f"{project_id}.{dataset_id}.predictions"
    predictions.to_gbq(output_table, project_id=project_id, if_exists='replace')
    print(f"Predictions saved to {output_table}")

if __name__ == "__main__":
    main()
