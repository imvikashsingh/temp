from pyspark.sql import SparkSession
from pyspark.sql.functions import col, pandas_udf, PandasUDFType
import joblib
from google.cloud import storage
import os
import tempfile
import pandas as pd

# Initialize Spark session with BigQuery support
spark = SparkSession.builder \
    .appName("LargeScaleBQMLPredictions") \
    .config("spark.jars.packages", "com.google.cloud.spark:spark-bigquery-with-dependencies_2.12:0.28.0") \
    .getOrCreate()

# Configuration
MODEL_BUCKET = "your-model-bucket"
MODEL_PATH = "path/to/model.bst"
BQ_PROJECT = "your-project"
BQ_DATASET = "your_dataset"
INPUT_TABLE = "input_table"
OUTPUT_TABLE = "predictions_output"

## Step 1: Load the model from GCS to all workers ##

def load_model_from_gcs():
    """Downloads and loads the model once per worker"""
    # Create temp directory
    temp_dir = tempfile.mkdtemp()
    local_path = os.path.join(temp_dir, "model.bst")
    
    # Download from GCS
    client = storage.Client()
    bucket = client.bucket(MODEL_BUCKET)
    blob = bucket.blob(MODEL_PATH)
    blob.download_to_filename(local_path)
    
    # Load model
    model = joblib.load(local_path)
    
    # Cleanup
    os.remove(local_path)
    os.rmdir(temp_dir)
    
    return model

# Broadcast the model loading function
model_broadcast = spark.sparkContext.broadcast(load_model_from_gcs)

## Step 2: Define prediction function ##

@pandas_udf("double", PandasUDFType.SCALAR)
def predict_udf(*cols):
    """Pandas UDF for making predictions"""
    # Get model from broadcast variable
    model = model_broadcast.value
    
    # Combine features into DataFrame
    features = pd.concat(cols, axis=1)
    
    # Make predictions
    predictions = model.predict(features)
    return pd.Series(predictions)

## Step 3: Load data from BigQuery ##

input_df = spark.read.format("bigquery") \
    .option("table", f"{BQ_PROJECT}.{BQ_DATASET}.{INPUT_TABLE}") \
    .load()

# Cache if dataset will be used multiple times
input_df.cache()

## Step 4: Prepare features and make predictions ##

# Get feature columns (adjust based on your model's requirements)
feature_cols = [col("feature1"), col("feature2"), col("feature3")]

# Add predictions to DataFrame
result_df = input_df.withColumn(
    "prediction",
    predict_udf(*feature_cols)
)

## Step 5: Save results back to BigQuery ##

result_df.write.format("bigquery") \
    .option("table", f"{BQ_PROJECT}.{BQ_DATASET}.{OUTPUT_TABLE}") \
    .option("temporaryGcsBucket", "your-temp-bucket") \
    .mode("overwrite") \
    .save()

print("Predictions completed and saved to BigQuery")
