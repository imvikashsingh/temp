from pyspark.sql import SparkSession
from pyspark.ml.feature import VectorAssembler
import pyspark.sql.functions as F
from google.cloud import storage
import xgboost as xgb
import numpy as np
from pyspark.sql.types import IntegerType

# Initialize Spark session with GCS and BigQuery configurations
spark = SparkSession.builder \
    .appName("XGBoostBoosterPrediction") \
    .config("spark.hadoop.fs.gs.impl", "com.google.cloud.hadoop.fs.gcs.GoogleHadoopFileSystem") \
    .config("spark.hadoop.fs.gs.impl.disable.cache", "true") \
    .config("spark.hadoop.google.cloud.auth.service.account.enable", "true") \
    .config("spark.hadoop.google.cloud.auth.service.account.json.keyfile", "/path/to/your-service-account-key.json") \
    .config("spark.jars", "/path/to/gcs-connector-hadoop3-latest.jar,/path/to/spark-bigquery-latest_2.12.jar") \
    .config("spark.datasource.bigquery.projectId", "your-gcp-project-id") \
    .getOrCreate()

# Parameters
model_path = "gs://your-bucket-name/path/to/model.bst"  # Update with your GCS model path
bigquery_table = "your-project-id.your_dataset.your_table"  # Update with your BigQuery table
output_path = "gs://your-bucket-name/path/to/output/predictions"  # Update with output path
bucket_name = "your-bucket-name"  # Update with your bucket name
model_file_path = "path/to/model.bst"  # Update with the path within the bucket
local_model_path = "/tmp/model.bst"  # Temporary local path for model

# Function to download model from GCS
def download_model_from_gcs(bucket_name, source_blob_name, destination_file_name):
    storage_client = storage.Client()
    bucket = storage_client.bucket(bucket_name)
    blob = bucket.blob(source_blob_name)
    blob.download_to_filename(destination_file_name)
    return destination_file_name

# Download model from GCS to local file
download_model_from_gcs(bucket_name, model_file_path, local_model_path)

# Load XGBoost model using Booster
xgb_model = xgb.Booster()
xgb_model.load_model(local_model_path)

# Broadcast model to Spark executors
broadcast_model = spark.sparkContext.broadcast(xgb_model)

# Load data from BigQuery
input_df = spark.read.format("bigquery") \
    .option("table", bigquery_table) \
    .load()

# Prepare features (modify feature_columns based on your data)
feature_columns = ["feature1", "feature2", "feature3"]  # Replace with actual feature column names
assembler = VectorAssembler(inputCols=feature_columns, outputCol="features")
feature_df = assembler.transform(input_df)

# UDF to predict using xgboost.Booster and DMatrix
def predict_udf(features):
    model = broadcast_model.value
    # Convert features to numpy array
    features_array = np.array(features)
    # Ensure 2D array for DMatrix
    if features_array.ndim == 1:
        features_array = features_array.reshape(1, -1)
    # Create DMatrix
    dmatrix = xgb.DMatrix(features_array)
    # Predict
    prediction = model.predict(dmatrix)[0]
    # For classification, convert to class label (assuming binary classification)
    return int(prediction > 0.5)  # Adjust threshold for your use case

predict_udf_spark = F.udf(predict_udf, IntegerType())

# Apply predictions
result_df = feature_df.withColumn("predicted_label", predict_udf_spark(F.col("features")))

# Select relevant columns (modify as needed)
result_df = result_df.select(
    "id",  # Replace with your identifier column
    "predicted_label"
)

# Save predictions to GCS
result_df.write.mode("overwrite").parquet(output_path)

# Stop Spark session
spark.stop()
